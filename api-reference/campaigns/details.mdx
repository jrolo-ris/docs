## Get Campaign Details

Retrieve detailed information about a specific video campaign.

<ParamField path="campaignId" type="string" required>
  The unique identifier of the campaign
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  Campaign details
</ResponseField>

<ResponseField name="data.id" type="string">
  Unique identifier for the campaign
</ResponseField>

<ResponseField name="data.name" type="string">
  Name of the campaign
</ResponseField>

<ResponseField name="data.description" type="string">
  Description of the campaign
</ResponseField>

<ResponseField name="data.status" type="string">
  Status of the campaign: `active`, `draft`, or `archived`
</ResponseField>

<ResponseField name="data.created_at" type="string">
  ISO timestamp of when the campaign was created
</ResponseField>

<ResponseField name="data.updated_at" type="string">
  ISO timestamp of when the campaign was last updated
</ResponseField>

<ResponseField name="data.views" type="integer">
  Total number of views for the campaign
</ResponseField>

<ResponseField name="data.unique_viewers" type="integer">
  Number of unique viewers for the campaign
</ResponseField>

<ResponseField name="data.average_watch_time" type="number">
  Average watch time in seconds
</ResponseField>

<ResponseField name="data.completion_rate" type="number">
  Percentage of viewers who watched the entire video
</ResponseField>

<ResponseField name="data.thumbnail_url" type="string">
  URL to the campaign thumbnail image
</ResponseField>

<ResponseField name="data.video_url" type="string">
  URL to the campaign video
</ResponseField>

<ResponseField name="data.settings" type="object">
  Campaign settings
</ResponseField>

<ResponseField name="data.settings.personalization" type="object">
  Personalization settings
</ResponseField>

<ResponseField name="data.settings.personalization.enabled" type="boolean">
  Whether personalization is enabled
</ResponseField>

<ResponseField name="data.settings.personalization.variables" type="array">
  Array of personalization variables
</ResponseField>

<ResponseField name="data.settings.sharing" type="object">
  Sharing settings
</ResponseField>

<ResponseField name="data.settings.sharing.enabled" type="boolean">
  Whether sharing is enabled
</ResponseField>

<ResponseField name="data.settings.sharing.platforms" type="array">
  Array of enabled sharing platforms
</ResponseField>

<ResponseField name="data.settings.tracking" type="object">
  Tracking settings
</ResponseField>

<ResponseField name="data.settings.tracking.enabled" type="boolean">
  Whether tracking is enabled
</ResponseField>

<ResponseField name="data.settings.tracking.events" type="array">
  Array of tracked events
</ResponseField>

## Example Request

```bash
curl -X GET "https://api.example.com/v1/campaigns/camp_1a2b3c4d5e6f" \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript
const axios = require('axios');

const apiKey = process.env.API_KEY;
const campaignId = 'camp_1a2b3c4d5e6f';

async function getCampaignDetails() {
  try {
    const response = await axios.get(`https://api.example.com/v1/campaigns/${campaignId}`, {
      headers: {
        'x-api-key': apiKey
      }
    });
    return response.data;
  } catch (error) {
    console.error('Error fetching campaign details:', error);
  }
}
```

```python
import requests
import os

api_key = os.environ.get('API_KEY')
campaign_id = 'camp_1a2b3c4d5e6f'

def get_campaign_details():
    headers = {
        'x-api-key': api_key
    }
    response = requests.get(f'https://api.example.com/v1/campaigns/{campaign_id}', 
                           headers=headers)
    response.raise_for_status()
    return response.json()
```

## Error Responses

<ResponseField name="success" type="boolean">
  Always `false` for error responses
</ResponseField>

<ResponseField name="error" type="object">
  Error details
</ResponseField>

<ResponseField name="error.code" type="string">
  Error code
</ResponseField>

</ResponseField>
