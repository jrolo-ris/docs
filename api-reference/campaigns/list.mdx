## List All Campaigns

Retrieve a paginated list of all your video campaigns, sorted by creation date.

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20" max="100">
  Number of campaigns to return per page
</ParamField>

<ParamField query="status" type="string">
  Filter campaigns by status: `active`, `draft`, `archived`
</ParamField>

<ParamField query="sort" type="string" default="-created_at">
  Sort campaigns by a specific field. Prefix with `-` for descending order. 
  Options: `created_at`, `updated_at`, `name`, `views`
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="array">
  Array of campaign objects
</ResponseField>

<ResponseField name="data[].id" type="string">
  Unique identifier for the campaign
</ResponseField>

<ResponseField name="data[].name" type="string">
  Name of the campaign
</ResponseField>

<ResponseField name="data[].status" type="string">
  Status of the campaign: `active`, `draft`, or `archived`
</ResponseField>

<ResponseField name="data[].created_at" type="string">
  ISO timestamp of when the campaign was created
</ResponseField>

<ResponseField name="data[].updated_at" type="string">
  ISO timestamp of when the campaign was last updated
</ResponseField>

<ResponseField name="data[].views" type="integer">
  Total number of views for the campaign
</ResponseField>

<ResponseField name="data[].thumbnail_url" type="string">
  URL to the campaign thumbnail image
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information
</ResponseField>

<ResponseField name="pagination.total" type="integer">
  Total number of campaigns
</ResponseField>

<ResponseField name="pagination.pages" type="integer">
  Total number of pages
</ResponseField>

<ResponseField name="pagination.page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="pagination.limit" type="integer">
  Number of campaigns per page
</ResponseField>

## Example Request

```bash
curl -X GET "https://api.example.com/v1/campaigns?page=1&limit=10&status=active" \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript
const axios = require('axios');

const apiKey = process.env.API_KEY;

async function listCampaigns() {
  try {
    const response = await axios.get('https://api.example.com/v1/campaigns', {
      params: {
        page: 1,
        limit: 10,
        status: 'active'
      },
      headers: {
        'x-api-key': apiKey
      }
    });
    return response.data;
  } catch (error) {
    console.error('Error listing campaigns:', error);
  }
}
```

```python
import requests
import os

api_key = os.environ.get('API_KEY')

def list_campaigns():
    headers = {
        'x-api-key': api_key
    }
    params = {
        'page': 1,
        'limit': 10,
        'status': 'active'
    }
    response = requests.get('https://api.example.com/v1/campaigns', 
                           headers=headers, 
                           params=params)
    response.raise_for_status()
    return response.json()
```
