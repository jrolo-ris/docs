## Get Viewers Information

Retrieve information about viewers who have watched your videos, including their engagement metrics and viewing history.

## Query Parameters

<ParamField query="email" type="string">
  Filter viewers by email address
</ParamField>

<ParamField query="video_id" type="string">
  Filter viewers by video ID
</ParamField>

<ParamField query="campaign_id" type="string">
  Filter viewers by campaign ID
</ParamField>

<ParamField query="start_date" type="string">
  Start date for the date range filter (ISO format: YYYY-MM-DD)
</ParamField>

<ParamField query="end_date" type="string">
  End date for the date range filter (ISO format: YYYY-MM-DD)
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20" max="100">
  Number of viewers to return per page
</ParamField>

<ParamField query="sort" type="string" default="-last_viewed_at">
  Sort viewers by a specific field. Prefix with `-` for descending order. 
  Options: `last_viewed_at`, `first_viewed_at`, `total_views`, `total_watch_time`
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="array">
  Array of viewer objects
</ResponseField>

</ResponseField>

<ResponseField name="data[].average_completion_rate" type="number">
  Average video completion rate (0-1)
</ResponseField>

<ResponseField name="data[].location" type="object">
  Location information of the viewer
</ResponseField>

<ResponseField name="data[].location.country" type="string">
  Country of the viewer
</ResponseField>

<ResponseField name="data[].location.city" type="string">
  City of the viewer
</ResponseField>

<ResponseField name="data[].location.region" type="string">

<ResponseField name="data[].device_info" type="object">

<ResponseField name="data[].device_info.type" type="string">
  Device type: `desktop`, `mobile`, `tablet`
</ResponseField>

<ResponseField name="data[].device_info.browser" type="string">
  Browser used by the viewer
</ResponseField>

<ResponseField name="data[].device_info.os" type="string">

</ResponseField>

<ResponseField name="pagination.page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="pagination.limit" type="integer">

## Example Request

```bash
curl -X GET "https://api.example.com/v1/analytics/viewers?campaign_id=camp_1a2b3c4d5e6f&start_date=2025-04-01&end_date=2025-04-10" \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript
const axios = require('axios');

const apiKey = process.env.API_KEY;

async function getViewers() {
  try {
    const response = await axios.get('https://api.example.com/v1/analytics/viewers', {
      params: {
        campaign_id: 'camp_1a2b3c4d5e6f',
        start_date: '2025-04-01',
        end_date: '2025-04-10'
      },
      headers: {
        'x-api-key': apiKey
      }
    });
    return response.data;
  } catch (error) {
    console.error('Error fetching viewers:', error);
  }
}
```

```python
import requests
import os

api_key = os.environ.get('API_KEY')

def get_viewers():
    headers = {
        'x-api-key': api_key
    }
    params = {
        'campaign_id': 'camp_1a2b3c4d5e6f',
        'start_date': '2025-04-01',
        'end_date': '2025-04-10'
    }
    response = requests.get('https://api.example.com/v1/analytics/viewers', 
                           headers=headers, 
                           params=params)
    response.raise_for_status()
    return response.json()
```

## Get Specific Viewer

To retrieve information about a specific viewer, you can use the email parameter:

```bash
curl -X GET "https://api.example.com/v1/analytics/viewers?email=john.doe@example.com" \
  -H "x-api-key: YOUR_API_KEY"
```

This will return the same response format but filtered to include only the specified viewer.
